* Rewrite History
#+BEGIN_SRC sh
  git filter-branch --commit-filter '
         if [ "$GIT_AUTHOR_EMAIL" = "schacon@localhost" ];
         then
                 GIT_AUTHOR_NAME="Scott Chacon";
                 GIT_AUTHOR_EMAIL="schacon@example.com";
                 git commit-tree "$@";
         else
                 git commit-tree "$@";
         fi' HEAD
#+END_SRC
